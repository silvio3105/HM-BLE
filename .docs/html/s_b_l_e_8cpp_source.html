55uint8_t sBLE_HM::init(void)
56{
57 // Return NOK if external handlers are NULL
58 if (!(TX && RX && sysKey && rst && status)) return SBLE_NOK;
59
60 // Ping module
61 if (ping() == SBLE_OK)
62 {
63 // Configure status output
64 sendCmd("AT+PIO11", 8);
65 //TX("AT+PIO11", 8);
66
67 // Return OK status
68 return SBLE_OK;
69 }
70
71 // Return NOK status
72 return SBLE_NOK;
73}

91uint8_t sBLE_HM::cmp(const char* input, const char* wanted, const uint8_t len)
92{
93 // Compare received answer with wanted answer
94 for (uint8_t i = 0; i < len; i++)
95 {
96 // Return NOK status if input is not equal to wanted
97 if (input[i] != wanted[i]) return SBLE_NOK;
98 }
99
100 return SBLE_OK;
101}

103uint8_t sBLE_HM::send(const char* tx, uint8_t txLen, const char* rx, const uint8_t rxLen)
104{
105 // Send command
106 TX(tx, txLen);
107
108 // Wait for answer if needed
109 if (rx)
110 {
111 // Receive module answer
112 RX(rxBuff, rxLen, '\0');
113
114 // Compare it and return compare status
115 return cmp(rxBuff, rx, rxLen);
116 }
117 else return SBLE_OK;
118}

120uint8_t sBLE_HM::sendReceive(const char* tx, const uint8_t txLen, const uint8_t rxLen, const char rxStop, char* output, const uint8_t rxOffset, const uint8_t rxMax)
121{
122 // Send command
123 TX(tx, txLen);
124
125 // Receive answer
126 RX(rxBuff, rxLen, rxStop);
127
128 // If first two chars are "OK"
129 if (rxBuff[0] == 'O' && rxBuff[1] == 'K' && rxBuff[2] == '+')
130 {
131 // Copy name to output variable
132 memcpy(output, &rxBuff[rxOffset], rxMax);
133
134 // Return OK status
135 return SBLE_OK;
136 }
137 else output[0] = '\0';
138
139 // Return NOK status
140 return SBLE_NOK;
141}

143uint8_t sBLE_HM::sendCmd(const char* cmd, ...)
144{
145 // Format command string
146 va_list args;
147 va_start(args, cmd);
148 uint8_t len = vsnprintf(rxBuff, sizeof(rxBuff), cmd, args);
149 va_end(args);
150
151 // Send formated command and return status
152 return send(rxBuff, len, "OK+", 3);
153}
